Swaggo:

download para o projeto:
    go get -u github.com/swaggo/swag/cmd/swag

gerar doc:
    swag init
    Flags para interepretar importações e dependencias (example; Gorm):
        swag init --parseDependency --parseInternal --parseDepth 1


Swagger editor:
    docker pull:
        docker pull swaggerapi/swagger-editor

    docker run:
        docker run -d -p 80:8080 swaggerapi/swagger-editor
        --or--
        You can provide your own json or yaml definition file from your local host:
            sudo docker run -p 80:8080 -v $(pwd):/tmp -e SWAGGER_FILE=/tmp/openapi.yaml swaggerapi/swagger-editor
                                        Pasta atual/ temporaria :  arquivoswagger.(yaml ou json)

Swagger UI:
    docker pull:
        docker pull swaggerapi/swagger-ui
        
    docker run:
        docker run -p 80:8080 swaggerapi/swagger-ui
        --or--    
            file direct ref (json):
            docker run -p 80:8080 -e SWAGGER_JSON=/tmp/openapi.json -v $(pwd):/tmp swaggerapi/swagger-ui
